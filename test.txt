class Animal()
    private string name
    private int age
    private string gender
    private string species
    private string color

    public new Animal(string name, int age, string gender, string species, string color)
        this.name = name
        this.age = age
        this.gender = gender
        this.species = species
        this.color = color
    end function   

    public string getName()
        return this.name
    end function 

    public void makeNoise()
        print("...")
    end function    

    end class


class Cat() extends Animal()
    private string breed

    public new Cat(string breed)
        super.new()
        this.breed = breed
    end function

    @Override
    public void makeNoise()
        print("MEOW")
    end function

end class

class HotelBooking()
    private int roomNo
    private string roomType
    private Currency cost
    private string customerName
    private Email customerEmail
    private Date startDate
    private Date endDate

    public new HotelBooking(int roomNo, Currency cost, string customerName, Email customerEmail, Date startDate, Date endDate)

        this.roomType = roomType
        this.cost = cost
        this.customerName = customerName
        this.customerEmail = customerEmail
        this.startDate = startDate
        this.endDate = endDate

    end function   

    public boolean stillStaying(Date date)
        if date >= this.endDate
            return false
        return true
    end function

end class
